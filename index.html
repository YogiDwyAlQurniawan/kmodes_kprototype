    • K-Modes¶

    ###### Source data : data.word-flag.data

    ###### Code :

    import numpy as np
from kmodes.kmodes import KModes

x = np.genfromtxt('flag.data.csv', dtype=int, delimiter=',')[:, :-1]

kmodes_huang = KModes(n_clusters=3, init='Huang', verbose=1)
kmodes_huang.fit(x)

print('k-modes (flag.data.csv) centroid:')
print(kmodes_huang.cluster_centroids_)
print('Final training cost: {}'.format(kmodes_huang.cost_))
print('Training Interations: {}'.format(kmodes_huang.n_iter_))

    • K-Prototype¶

    ###### Source data : data.word-zoo.data

    ###### Code :

    import numpy as np
from kmodes.kprototypes import KPrototypes

syms = np.genfromtxt('zoo.data.csv', dtype=str, delimiter=',')[:, 0]
X = np.genfromtxt('zoo.data.csv', dtype=object, delimiter=',')[:, 1:]
X[:, 0] = X[:, 0].astype(float)

kproto = KPrototypes(n_clusters=6, init='Cao', verbose=1)
clusters = kproto.fit_predict(X, categorical=[9, 2])

print(kproto.cluster_centroids_)
print(kproto.cost_)
print(kproto.n_iter_)

for s, c in zip(syms, clusters):
    print("Symbol: {}, cluster:{}".format(s, c))
